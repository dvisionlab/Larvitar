Press "p" to play/pause timeserie animation 

          
          

            let demoFiles = [];
            let counter = 0;

            const getDemoFileNames = function () {
              let demoFileList = [];
              for (let i = 1; i < 195; i++) {
                let filename = "image_(" + i + ").dcm";
                demoFileList.push(filename);
              }
              return demoFileList;
            };

            // init all
            larvitar.initializeImageLoader();
            larvitar.initializeCSTools();
            larvitar.store.initialize();
            larvitar.store.addViewport("viewer");

            async function createFile(fileName, cb) {
              let response = await fetch("./demo/4d/" + fileName);
              let data = await response.blob();
              let file = new File([data], fileName);
              demoFiles.push(file);
              counter++;
              if (counter == 194) {
                cb();
              }
            }

            function renderSerie() {
              larvitar.resetLarvitarManager();
              larvitar
                .readFiles(demoFiles)
                .then(seriesStack => {
                  // render the first series of the study
                  let sliceId = 0;
                  let seriesId = _.keys(seriesStack)[sliceId];
                  let serie = seriesStack[seriesId];

                  larvitar.renderImage(serie, "viewer").then(() => {
                    console.log("Image has been rendered");

                    $("#image-time").html(
                      "Image Time Id: " +
                        larvitar.store.get(["viewports", "viewer", "timeId"]) +
                        " of " +
                        larvitar.store.get(["viewports", "viewer", "maxTimeId"])
                    );
                    $("#timestamp").html(
                      "Image Time: " +
                        larvitar.store.get(["viewports", "viewer", "timestamp"])
                    );
                  });

                  let animation = false;
                  let animationId = null;
                  document.onkeypress = function (e) {
                    if (e.keyCode == 112) {
                      animation = !animation;
                      if (animation) {
                        sliceId = larvitar.store.get([
                          "viewports",
                          "viewer",
                          "sliceId"
                        ]);
                        animationId = setInterval(function () {
                          sliceId = sliceId == 193 ? 0 : sliceId + 1;
                          larvitar.updateImage(serie, "viewer", sliceId, false);
                          $("#image-time").html(
                            "Image Time Id: " +
                              larvitar.store.get(["viewports", "viewer", "timeId"]) +
                              " of " +
                              larvitar.store.get(["viewports", "viewer", "maxTimeId"])
                          );
                          $("#timestamp").html(
                            "Image Time: " +
                              larvitar.store.get(["viewports", "viewer", "timestamp"])
                          );
                        }, 100);
                      } else {
                        clearInterval(animationId);
                      }
                    }
                  };
                  larvitar.addDefaultTools();
                  larvitar.setToolActive("Wwwc");
                })
                .catch(err => console.log(err));
            }

            let demoFileList = getDemoFileNames();
            _.each(demoFileList, function (demoFile) {
              createFile(demoFile, renderSerie);
            });
