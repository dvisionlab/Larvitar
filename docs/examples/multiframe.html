            let demoFiles = [];

            // init all
            larvitar.initLarvitarStore();
            larvitar.initializeImageLoader();
            larvitar.registerMultiFrameImageLoader();
            larvitar.initializeCSTools();
            larvitar.larvitar_store.addViewport("viewer");

            function renderSerie() {
              larvitar.resetImageParsing();
              larvitar.readFiles(demoFiles, function (seriesStack, err) {
                // render the first series of the study
                let seriesId = _.keys(seriesStack)[0];
                let serie = seriesStack[seriesId];
                larvitar.buildMultiFrameImage(seriesId, serie);
                let multiFrameSerie = larvitar.multiFrameManager[seriesId];
                larvitar.renderFrame(multiFrameSerie, "viewer");
                // // larvitar.renderImage(multiFrameSerie, "viewer");
                larvitar.addDefaultTools();
                larvitar.setToolActive(
                  larvitar.larvitar_store.state.leftMouseHandler
                );
                

                let animation = false;
                let index = 0;
                let animationId = null;

                document.onkeypress = function (e) {
                  e = e || window.event;
                  
                  if (e.keyCode == 112) {
                    animation = !animation;
                    if (animation) {
                      animationId = setInterval(function () {
                        index = index == 75 ? 0 : index + 1; // numberOfFrames = 76
                        larvitar.renderFrame(multiFrameSerie, "viewer", index);
                        $("#current-frame").html("Current Frame: " + parseInt(index+1) + " of 76");
                      }, 80); // frame rate
                    } else {
                      clearInterval(animationId);
                    }
                  }
                };
              });
            }

            async function createFile(fileName, cb) {
              let response = await fetch("./demo/" + fileName);
              let data = await response.blob();
              let file = new File([data], fileName);
              demoFiles.push(file);
              cb();
            }
            createFile("xa_integris.dcm", renderSerie);
