//CUSTOM TOOL 
const cornerstoneTools = larvitar.cornerstoneTools;
const MagnifyTool = cornerstoneTools.MagnifyTool;


class MyMagnifyTool extends MagnifyTool {
  constructor(props) {
    super(props);
  }

  /**
   * Overrides the _drawMagnificationTool method to add red borders.
   * @param {*} evt
   * @returns {void}
   */
  _drawMagnificationTool(evt) {
    super._drawMagnificationTool(evt); // Call the parent method to draw the magnification tool

    const magnifyCanvas = evt.detail.element.querySelector('.magnifyTool');
    //console.log(magnifyCanvas);
    if (magnifyCanvas) {
      //console.log("you are here");
      const context = magnifyCanvas.getContext('2d');

      // Add red borders
      context.strokeStyle = 'red';
      context.lineWidth = 2;
      context.strokeRect(0, 0, magnifyCanvas.width, magnifyCanvas.height);
    }
  }
}

              let response = await fetch("./demo/" + fileName);

              let file = new File([data], fileName);

            function renderSerie() {
              larvitar.resetLarvitarManager();
              larvitar
